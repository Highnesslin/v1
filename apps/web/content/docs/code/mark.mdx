## !implementation

We could do this in a few ways. One way is to have two components, one for _marked_ lines and one for lines that are not marked.

<CodeWithNotes>

```tsx ! code.tsx
// !Collapse(1:6) collapsed
import {
  RawCode,
  Pre,
  highlight,
  LineAnnotationComponent,
  LineComponent,
} from "codehike/code"

export async function Code({ codeblock }: { codeblock: RawCode }) {
  const highlighted = await highlight(codeblock, "github-dark")
  return (
    <Pre
      className="m-0 px-0 bg-zinc-950"
      code={highlighted}
      components={{ Line, LineMark }}
    />
  )
}

const Line: LineComponent = ({ children }) => {
  // !Callout[/border-t/] transparent left-border for normal lines
  return <div className="px-2 border-l-2 border-transparent">{children}</div>
}

const LineMark: LineAnnotationComponent = ({ children }) => {
  return (
    // !Callout[/border-blue-400/] blue border and background for marked lines
    <div className="px-2 border-l-2 border-blue-400 bg-blue-400/10">
      {children}
    </div>
  )
}
```
